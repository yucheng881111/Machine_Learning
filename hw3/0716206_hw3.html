fig, ax = plt.subplots()
xI=[]
yI=[]
for i in range(1000):
    yi=random.choice([0,1])
    x, y = np.random.multivariate_normal([yi,yi], [[0.1,0],[0,0.1]], 1).T
    xI.append([x[0],y[0]])
    yI.append(yi)
    if yi==1:
        s1=ax.scatter(x, y, color='red')
    else:
        s2=ax.scatter(x, y, color='blue')
plt.xlabel(r"$x_{i0}$")
plt.ylabel(r"$x_{i1}$")
plt.legend((s2,s1),(r"$y_{i}=0$",r"$y_{i}=1$"), loc=2, scatterpoints=1)
plt.show()
